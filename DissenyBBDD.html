Disseny Físic

Sentències CREATE TABLE

  • CREATE TABLE PERSONA (
        id_persona INT AUTO_INCREMENT PRIMARY KEY,
        nom VARCHAR(50) NOT NULL,
        cognom VARCHAR(50) NOT NULL,
        adreça VARCHAR(100) NOT NULL,
        doc_identitat VARCHAR(20) UNIQUE NOT NULL,
        data_naixement DATE NOT NULL,
        telefon VARCHAR(15) NOT NULL,
        email VARCHAR(100) NOT NULL
    );
  • CREATE TABLE EMPLEAT (
        id_empleat INT PRIMARY KEY,
        lloc_feina VARCHAR(50) NOT NULL,
        data_contractacio DATE NOT NULL,
        salari_brut DECIMAL(10, 2) NOT NULL,
        estat_laboral VARCHAR(20) NOT NULL,
        FOREIGN KEY (id_empleat) REFERENCES PERSONA(id_persona)
    );
  • CREATE TABLE TASCA (
        id_tasca INT AUTO_INCREMENT PRIMARY KEY,
        descripcio VARCHAR(200) NOT NULL,
        data_creacio DATE NOT NULL,
        data_execucio DATE CHECK (data_execucio >= data_creacio),
        estat_tasca VARCHAR(20) NOT NULL
    );
  • CREATE TABLE REALITZAR (
        id_empleat INT NOT NULL,
        id_tasca INT NOT NULL,
        estat_realitzacio VARCHAR(20) NOT NULL,
        PRIMARY KEY (id_empleat, id_tasca),
        FOREIGN KEY (id_empleat) REFERENCES EMPLEAT(id_empleat),
        FOREIGN KEY (id_tasca) REFERENCES TASCA(id_tasca)
    );
  • CREATE TABLE CLIENT (
        id_client INT PRIMARY KEY,
        data_registre DATE NOT NULL,
        tipus_client VARCHAR(50) NOT NULL,
        targeta_credit VARCHAR(16) NOT NULL,
        FOREIGN KEY (id_client) REFERENCES PERSONA(id_persona)
    );
  • CREATE TABLE HABITACIO (
        id_habitacio INT AUTO_INCREMENT PRIMARY KEY,
        num_habitacio INT NOT NULL,
        tipus VARCHAR(50) NOT NULL,
        capacitat INT NOT NULL,
        preu_nit_AD DECIMAL(10, 2) NOT NULL,
        preu_nit_MP DECIMAL(10, 2) NOT NULL,
        estat_habitacio VARCHAR(20) NOT NULL,
        descripcio VARCHAR(200) NOT NULL
    );
  • CREATE TABLE RESERVA (
        id_reserva INT AUTO_INCREMENT PRIMARY KEY,
        data_reserva DATE NOT NULL,
        data_inici DATE NOT NULL CHECK (data_inici >= data_reserva),
        data_fi DATE NOT NULL CHECK (data_fi > data_inici),
        tipus_reserva VARCHAR(50) NOT NULL,
        tipus_iva DECIMAL(5, 2) NOT NULL,
        preu_total DECIMAL(10, 2) NOT NULL,
        id_client INT NOT NULL,
        id_habitacio INT NOT NULL,
        FOREIGN KEY (id_client) REFERENCES CLIENT(id_client),
        FOREIGN KEY (id_habitacio) REFERENCES HABITACIO(id_habitacio)
    );
  • CREATE TABLE FACTURA (
        id_factura INT AUTO_INCREMENT PRIMARY KEY,
        data_emisio DATE NOT NULL,
        metode_pagament VARCHAR(50) NOT NULL,
        base_imposable DECIMAL(10, 2) NOT NULL,
        iva DECIMAL(10, 2) NOT NULL,
        total DECIMAL(10, 2) NOT NULL,
        id_reserva INT NOT NULL,
        FOREIGN KEY (id_reserva) REFERENCES RESERVA(id_reserva)
    );
